# Identifique a finalidade dos seguintes comandos:
    a. git init --> INICIA UM REPOSITÓRIO LOCAL
    b. git config --global user.name "turing" --> CONFIGURA, DE FORMA GLOBAL, O NOME DO USUÁRIO
    c. git add EXERCICIO.txt --> TRANSFERE O ARQUIVO EXERCICIO.TXT PARA ETAPA DE STAGED
    d. git add -A - TRANSFERE TODOS OS ARQUIVOS PARTINDO DO DIRETÓRIO RAIZ PARA ETAPA DE STAGED
    e. git commit -m "Adicionado nova interface" --> SOBE TODOS OS ARQUIVOS QUE ESTÃO EM STAGED PARA O REPOSITÓRIO LOCAL, ADICIONANDO A MENSAGEM “Adicionado nova interface”
    f. git commit --> SOBE TODOS OS ARQUIVOS QUE ESTÃO EM STAGED PARA O REPOSITÓRIO LOCAL SEM NENHUMA MENSAGEM
    g. git reset --hard HEAD --> DESFAZ AS ALTERAÇÕES DESDE O ÚLTIMO COMMIT
    h. cd Downloads --> MUDA PARA O DIRETÓRIO DOWNLOADS
    i. pwd --> DIZ QUAL O DIRETÓRIO ATUAL
    j. cd .. --> VOLTA UMA PASTA
    k. ls --> LISTA OS ITENS DE UM DIRETÓRIO
    l. git pull --> ATUALIZA O REPOSITÓRIO LOCAL COM OS ARQUIVOS DO REPOSITÓRIO REMOTO
    m. git push --> SOBE OS ARQUIVOS DO REPOSITÓRIO LOCAL PARA O REPOSITÓRIO REMOTO
    n. git clone <<url>> --> CLONA/BAIXA OS ARQUIVOS DE UM REPOSITÓRIO REMOTO INFORMADO
    o. git diff --> COMPARA A DIFERENÇAS ENTRE O REPOSITÓRIO LOCAL E O REMOTO
    p. git show --> LISTA OS ÚLTIMOS COMMITS E INFORMAÇÕES SOBRE ELES


# Descreva a função dos seguintes componentes do Git
    a. Working Tree, Working Area --> É ESTRUTURA DE DIRETÓRIOS E ARQUIVOS, ONDE OS ARQUIVOS SÃO EDITADOS, ADICIONADOS E/OU EXCLUÍDOS.
    b. Stage Area, Commit. --> COMMIT É RESPONSÁVEL POR ENVIAR OS ARQUIVOS OBSERVADOS PARA O REPOSITÓRIO LOCAL. OS ARQUIVOS NA ÁREA DE STAGED SÃO OS ARQUIVOS QUE SÃO “OBSERVADOS” PELO GIT.
    c. Local Repository --> É O REPOSITÓRIO LOCAL, LOCALIZAÇÃO DOS ARQUIVOS APÓS REALIZADOS OS COMMITS.
    d. Remote Repository --> É O REPOSITÓRIO LOCAL, REPOSITÓRIO EM QUE OS ARQUIVOS SÃO DIRECIONADOS APÓS SAIR DO REPOSITÓRIO LOCAL. É ONDE AS VERSÃO “COMPLETAS” SE ENCONTRAM.
